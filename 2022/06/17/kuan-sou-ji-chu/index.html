如题：BFS基础

宽搜BFS基础篇

最近简单的学习了一下搜索的相关知识，鉴于本人是个健忘症患者，有必要写个总结来时刻告诉自己是个笨比。

宽度优先搜索BFS

BFS的特点是按层读入，即将当前状态的所有可能的次态纳入考虑范围，这样层层递进，根据这个特点，所有状态第一次被搜到时都是距离起点最近的。

为了更直观的感受，举一个简单的例子，问一个N×N的矩阵中的某一点到其他个点的最短距离（只允许上下左右移动）。

BFS的实现，BFS需要用到的数据结构——队列，正因为队列的特性先进先出，使得BFS的路径一定最短。

基本模板

void bfs()
{
    queue<type>q;
    q.push(第一层);
    // 标记第一层
    while(!q.empty()) {
        // 取出队头
        auto t = q.front();
        q.pop();
        /*
        进行操作, 目的是将所有符合条件的下一层入队
        */
    }
}

BFS的枚举次数很少，一般不会TLE，如果有就是你BFS写错了。

{
	queue<PII> q;
//	memset(st, false, sizeof st);
	q.push({x,y});
	st[x][y] = true;
	int cnt = 1;
	vector<PII> res;
	res.push_back({x,y});
	while(q.size()) {
		auto t = q.front();
		q.pop();
		for(int i = 0;i < 4;i ++) {
			int a = t.first + dx[i], b = t.second + dy[i];
			if(a >= 1 && a <= n && b >= 1 && b <= n && !st[a][b] && (g[a][b] ^ g[t.first][t.second])) {

			}
		}

	}

P1162 填涂颜色

这题和细胞个数那一题非常像，不过需要逆向思考，求完全闭包连通块，只要由边界出发，将边界得到的连通块标记ok

AC代码

#include <iostream>
#include <cstring>
#include <queue>

using namespace std;
typedef pair<int,int> PII;

int n;
int g[50][50];
bool st[50][50];
int dx[4] = {-1, 1, 0, 0}, dy[4] = {0, 0, -1, 1};

void bfs(int x,int y)
{
	queue<PII>q;
	q.push({x,y});
	st[x][y] = true;
	
	while(q.size()) {
		auto t = q.front();
		q.pop();
		
		for(int i = 0;i < 4;i ++) {
			int a = t.first + dx[i], b = t.second + dy[i];
			if(a >= 1 && a <= n && b >= 1 && b <= n && !st[a][b] && g[a][b] == 0) {
				q.push({a,b});
				st[a][b] = true;
			}
		}
	}
}

int main()
{
	cin >> n;
	
	for(int i = 1;i <= n;i ++) for(int j = 1;j <= n;j ++) cin >> g[i][j];
	
	// 找出所有边界上的 0 的连通块
	
	for(int i = 1;i <= n;i ++) {
		for(int j = 1;j <= n;j ++) {
			if((i == 1 || i == n || j == 1 || j == n) && g[i][j] == 0 && !st[i][j]) {
				bfs(i,j);
			}
		}
	}
	
	for(int i = 1;i <= n;i ++) {
		for(int j = 1;j <= n;j ++) {
			if(!st[i][j] && g[i][j] == 0) {
				cout << "2 ";
			}
			else cout << g[i][j] << " ";
		}
		cout << endl;
	}
	return 0;
}

P2360 地下城主

这道题的亮点在于，我的移动方法要开成三维的，dx, dy,dz;

适用于在多个维度跨层宽搜

AC代码

#include <iostream>
#include <cstring>
#include <queue>

using namespace std;

int l, c, r;
int beginx, beginy, begin_l, endx, endy, end_l;
string g[40][40];
//bool st[40][40][40];
int d[40][40][40];
int dx[]={1,-1,0,0,0,0},dy[]={0,0,1,-1,0,0},dz[]={0,0,0,0,1,-1};

struct node {
	int x,y,z;
	
};

int bfs(int x, int y, int z) {
	
	queue<pair<int,node>> q;
	
	q.push({0,{x,y,z}});
	
	memset(d,-1,sizeof d);
	
	d[z][x][y] = 0;
	
	while(q.size()) {
		auto t = q.front();
		q.pop();
		
		int distance = t.first;
		int a = t.second.x, b = t.second.y, m = t.second.z;
		
		for(int i = 0;i < 6;i ++) {
			int u = a + dx[i], v = b + dy[i], w = m + dz[i];
			// u 是行， v 是列， w 是层数
			if(u >= 0 && u < r && v >= 0 && v < c && w >= 0 && w < l && d[w][u][v] == -1 && g[w][u][v] != '#') {
				q.push({distance + 1, {u,v,w}});
				d[w][u][v] = distance + 1;
				if(g[w][u][v] == 'E') return d[w][u][v];
			}
		}
	}
	return -1;
}

int main() {
	cin >> l >> r >> c;
	for (int i = 0; i < l; i ++) {
		for (int j = 0; j < r; j ++) {
			cin >> g[i][j];
			int p1 = g[i][j].find("S");
			int p2 = g[i][j].find("E");
			if (p1 != g[i][j].npos) {
				beginx = j, beginy = p1, begin_l = i;
			}
			if (p2 != g[i][j].npos) {
				endx = j, endy = p2, end_l = i;
			}
		}
	}
	
	int t = bfs(beginx,beginy,begin_l);
	
	if(t == -1) cout << "Trapped!" << endl;
	else cout << "Escaped in " << t << " minute(s)." << endl;
	
	return 0;
}

			q.pop();
			int dist = tt.dist, x = tt.x, y = tt.y;
			for (int i = 1; i <= 4; i ++)
				{
					int ft = abs(i - tt.f);
					if (ft % 2 == 0 && ft != 0) ft = 2;
					else if (ft != 0) ft = 1;

					for (int j = 0; j < 3; j ++)
						{
							int now_d = dx[j];
							if (i == 1)
								{
									// dong
									int next = y + now_d;
									if (next > m - 1) continue;
									if (now_d == 3 && d[x][next - 1] == inf) continue;
									if (d[x][next] == -1 || (d[x][next] != inf && d[x][next] > dist + ft + 1))
										{
											d[x][next] = dist + ft + 1;
											q.push({dist + ft + 1, x, next, i});
										}
								}
							else if (i == 2)
								{
									// nan
									int next = x + now_d;
									if (next > n - 1) continue;
									if (now_d == 3 && d[next - 1][y] == inf) continue;
									if (d[next][y] == -1 || (d[next][y] != inf && d[next][y] > dist + ft + 1))
										{
											d[next][y] = dist + ft + 1;
											q.push({dist + ft + 1, next, y, i});
										}
								}
							else if (i == 3)
								{
									// xi
									int next = y - now_d;
									if (next < 1) continue;
									if (now_d == 3 && d[x][next + 1] == inf) continue;
									if (d[x][next] == -1 || (d[x][next] != inf && d[x][next] > dist + ft + 1))
										{
											d[x][next] = dist + ft + 1;
											q.push({dist + ft + 1, x, next, i});
										}
								}
							else
								{
									// bei
									int next = x - now_d;
									if (next < 1) continue;
									if (now_d == 3 && d[next + 1][y] == inf) continue;
									if (d[next][y] == -1 || (d[next][y] != inf && d[next][y] > dist + ft + 1))
										{
											d[next][y] = dist + ft + 1;
											q.push({dist + ft + 1, next, y, i});
										}
								}
						}
				}
		}
}

int main()
{
	memset(d, -1, sizeof d);
	cin >> n >> m;
	for (int i = 1; i <= n; i ++)
		{
			for (int j = 1; j <= m; j ++)
				{
					cin >> g[i][j];
					if (g[i][j])
						{
							d[i][j - 1] = d[i - 1][j - 1] = d[i - 1][j] = d[i][j] = inf;
						}
				}
		}
	int a, b, c, e;
	string s;
	cin >> a >> b >> c >> e >> s;

	if (s == "E") f = 1;
	else if (s == "S") f = 2;
	else if (s == "W") f = 3;
	else f = 4;
    
	bfs(a, b);
    
	if(d[c][e] != inf) cout << d[c][e] << endl;

}
