质数筛-欧拉筛（线性筛）

传送门 3284. 猫猫藏宝图

#include<iostream>
#define MaxN 10001;

using namespace std;

bool vis[MaxN],isprime[MaxN]
int cnt;
void Prime(int n)
{
    for(int i = 2;i <= n;i ++)
    {
        if(!vis[i]) vis[i] = true,isprime[cnt++] = i; // 这里 可以添加 isPrime[i] = true,作标记素组
        for(int j = 0;j < cnt;j ++)
        {
            if(i * isprime[j] > n) break;
            vis[i * isprime[j]] = true;
            if(i % isprime[j] == 0) break;
        }
    }
    // jieshu
}
int main()
{
    int n;
    while(n --)
    {
        int a;
        cin >> a;
        if(isPrime[a]) cout << a << "is a prime" << endl;
        else cout << a << "isn't a prime" << endl;
    }
    return 0;
}
